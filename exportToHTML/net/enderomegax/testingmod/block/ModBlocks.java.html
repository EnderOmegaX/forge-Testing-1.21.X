15   import java.util.function.Supplier;

17   public class ModBlocks {
18       public static final DeferredRegister<Block> BLOCKS = DeferredRegister.create(ForgeRegistries.BLOCKS, TestingMod.MOD_ID);
19   
20   
21       public static final RegistryObject<Block> UNEARTHED_FOSSIL = registerBlock("unearthed_fossil", () -> new Block(BlockBehaviour.Properties.of()
22               .strength(4f).requiresCorrectToolForDrops().sound(SoundType.BONE_BLOCK)));
23   
24   
25   
26       private static <T extends Block> RegistryObject<T> registryObject(String name, Supplier<T> block) {
27           RegistryObject<T> toReturn = BLOCKS.register(name, block);
28           registerBlockItem(name, toReturn);
29           return toReturn;
30       }
31   
32       private static <T extends Block> void registerBlockItem(String name, RegistryObject<T> block) {
33           Moditems.ITEMS.register(name, () -> new BlockItem(block.get(), new Item.Properties()));
34       }
35   
36   
37       public static void register(IEventBus eventBus){
38           BLOCKS.register(eventBus);
39       }
